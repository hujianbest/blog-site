```typescript
// utils/validation.ts
export function validateFieldName(
  value: any,
  fieldName: string,
  maxLength: number = 50
): ValidationResult {
  if (!value) {
    return { valid: false, error: `${fieldName} is required` };
  }
  if (value.trim().length === 0) {
    return { valid: false, error: `${fieldName} cannot be empty` };
  }
  if (value.length > maxLength) {
    return { valid: false, error: `${fieldName} must be ${maxLength} characters or less` };
  }
  return { valid: true };
}

export function sanitizePaginationParams(limit?: any, offset?: any): {
  take?: number;
  skip?: number;
} {
  const result: { take?: number; skip?: number } = {};
  if (limit !== undefined && limit !== null) {
    const limitNum = parseInt(limit as string);
    if (!isNaN(limitNum)) {
      result.take = Math.min(Math.max(0, limitNum), 100);
    }
  }
  if (offset !== undefined && offset !== null) {
    const offsetNum = parseInt(offset as string);
    if (!isNaN(offsetNum)) {
      result.skip = Math.max(0, offsetNum);
    }
  }
  return result;
}
```

**使用示例**:
```typescript
// tags.controller.ts
const validation = validateFieldName(name, 'Tag name');
if (!validation.valid) {
  return res.status(400).json({
    error: { message: validation.error }
  });
}
```

**效果**:
- ✅ 代码去重
- ✅ 统一验证逻辑
- ✅ 易于维护和扩展

**文件**: `tags.controller.ts:72-84`

**原始问题**:
```typescript
const { limit, offset } = req.query;
const tags = await prisma.tag.findMany({
```

```typescript
  skip: pagination.skip,    // ✅ 最小 0
```
